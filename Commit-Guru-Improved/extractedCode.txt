class LocalRepository():
    repo = None
    adapter = None
    start_date = None
    def __init__(self, repo):
        self.repo = repo
        self.adapter = Git
        self.commits = {}
    def sync(self):
        firstSync = self.syncRepoFiles()
        self.syncCommits(firstSync)
        self.repo.ingestion_date = self.start_date
    def syncRepoFiles(self):
        self.start_date = str(datetime.now().replace(microsecond=0))
        path = os.path.dirname(__file__) + self.adapter.REPO_DIRECTORY + self.repo.id
        if os.path.isdir(path):
            self.adapter.pull(self.adapter, self.repo)
            firstSync = False
        else:
            self.adapter.clone(self.adapter, self.repo)
            firstSync = True
        return firstSync
    def syncCommits(self, firstSync):
        commits = self.adapter.log(self.adapter, self.repo, firstSync)
        commitsSession = Session()
        logging.info('Saving commits to the database...')
        for commitDict in commits:
            commitDict['repository_id'] = self.repo.id
            commitsSession.merge(Commit(commitDict))
        commitsSession.commit()
        commitsSession.close()
        logging.info('Done saving commits to the database.')

import copy
class Commit:
    def __init__(self) -> None:
        self.hash_code=''
        self.parent_hash=None
        self.file_dict=None
        self.Is_merge=False
        self.Is_branch_start=False
        self.followCommit=[]
        self.copy_count=0
    def get_dict(self)->dict:
        if self.Is_branch_start==True:
            return copy.deepcopy(self.file_dict)
        else:
            return self.file_dict
    def update_dict(self,file_dict:dict):
        if self.copy_count==len(self.followCommit):
            file_dict.clear()
            return
        if self.Is_merge==True:
            self.file_dict=copy.deepcopy(file_dict)
        else:
            self.file_dict=file_dict
    def release_memory(self):
        self.copy_count+=1
        if self.copy_count==len(self.followCommit):
            if self.Is_branch_start:
                self.file_dict.clear()
            self.file_dict=None

def ingestRepo(repository_to_ingest, session):
  logging.info( 'A worker is starting scan repository: ' +
                      repository_to_ingest.id )
  repository_to_ingest.status = "Ingesting"
  session.commit()
  local_repo = LocalRepository(repository_to_ingest)
  local_repo.sync()
  session.merge(repository_to_ingest) 
  repository_to_ingest.status = "Waiting to be Analyzed" 
  session.commit() 
  logging.info( 'A worker finished ingesting repo ' + 
                  repository_to_ingest.id )
  session.close()
def ingest(repo_id):
  session = Session()
  repo_to_analyze = (session.query(Repository)
        .filter (Repository.id == repo_id)
        .all()
        )
  if len(repo_to_analyze) == 1:
  	ingestRepo(repo_to_analyze[0], session)
  else:
    logging.info('Repo with id ' + repo_id_to_analyze + ' not found!')
  session.close()

    def clone(self, repo):
        repo_dir = os.chdir(os.path.join(os.path.dirname(__file__),self.REPO_DIRECTORY))
        logging.info('Git cloning repo: '+ str(repo) )
        cloneResult = str(subprocess.check_output(
                  self.CLONE_CMD.format(repo.url, './' + repo.id),
                  shell= True,
                  cwd = repo_dir ) )
        logging.info('Done cloning.')
        return True
    def pull(self, repo):
        repo_dir = os.path.dirname(__file__) + self.REPO_DIRECTORY + repo.id
        subprocess.call(self.RESET_CMD, shell=True, cwd= repo_dir)
        subprocess.call(self.CLEAN_CMD, shell=True, cwd= repo_dir)
        logging.info('Pulling latest changes from repo: '+ str(repo) )
        fetchResult = str(subprocess.check_output(
                  self.RESET_CMD + "\n" + self.PULL_CMD ,
                  shell=True,
                  cwd=  repo_dir  ) )
        logging.info('Done fetching.')
        return True

import csv 
class Category():
	associatedWords = [] 
	category_name = None 
	def __init__(self, fileLocation, name):
		self.category_name = name
		self.associatedWords = [] 
		self.readInAssociatedWords(fileLocation)
	def readInAssociatedWords(self, fileLocation):
		with open(fileLocation, 'rt',encoding="utf-8") as csvfile:
			wordreader = csv.reader(csvfile, delimiter=',', quotechar='|')
			for row in wordreader:
				for word in row:
					self.associatedWords.append(word)
	def belongs(self, commit_msg):
		commit_msg = commit_msg.lower()  
		for assoc_word in self.associatedWords:
			assoc_word = assoc_word.split("...")  
			found_keyword = True
			for part_keyword in assoc_word: 
				if found_keyword and part_keyword in commit_msg:
					found_keyword = True
				else:
					found_keyword = False
			if found_keyword:
				return True
		return False
	def getName(self):
		return self.category_name

class Worker(threading.Thread):
	def __init__(self, tasks):
		threading.Thread.__init__(self)
		self.tasks = tasks
		self.daemon = True
		self.start()
	def run(self):
		while True:
			func, args, kargs = self.tasks.get()
			try:
				func(*args, **kargs)
			except Exception as e:
				print(e)
			self.tasks.task_done()
class ThreadPool:
	def __init__(self, num_threads):
		self.tasks = Queue(num_threads)
		for _ in range(num_threads): Worker(self.tasks)
	def add_task(self, func, *args, **kargs):
		self.tasks.put((func, args, kargs))
	def wait_completion(self):
		self.tasks.join()

    def getCommitStatsProperties(self,stats,commitFiles, devExperience,hash_code,parents, author, unixTimeStamp ):
        statProperties = ""
        subsystemsSeen = []                         
        directoriesSeen = []                        
        locModifiedPerFile = []                     
        authors = []                                
        fileAges = []                               
        la = 0                                      
        ld = 0                                      
        nf = 0                                      
        ns = 0                                      
        nd = 0                                      
        entrophy = 0                                
        lt = 0                                      
        ndev = 0                                    
        age = 0                                     
        exp = 0                                     
        rexp = 0                                    
        sexp = 0                                    
        totalLOCModified = 0                        
        nuc = 0                                     
        filesSeen = ""                              
        for stat in stats:
            if( stat == ' ' or stat == '' ):
                continue
            fileStat = stat.split("\t") 
            if( len(fileStat) < 2):
                continue
            try:
                fileLa = int(fileStat[0])
                fileLd = int(fileStat[1])
            except:
                fileLa = 0
                fileLd = 0
            fileName = (fileStat[2].replace("'",'').replace('"','').replace("\\",""))
            totalModified = fileLa + fileLd
            if(fileName in commitFiles):
                prevFileChanged = commitFiles[fileName]
                prevLOC = getattr(prevFileChanged, 'loc')
                prevAuthors = getattr(prevFileChanged, 'authors')
                prevChanged = getattr(prevFileChanged, 'lastchanged')
                file_nuc = getattr(prevFileChanged, 'nuc')
                nuc += file_nuc
                lt += prevLOC
                for prevAuthor in prevAuthors:
                    if prevAuthor not in authors:
                        authors.append(prevAuthor)
                age += ( (int(unixTimeStamp) - int(prevChanged)) / 86400 )
                fileAges.append(prevChanged)
                file_nuc += 1 
                setattr(prevFileChanged, 'loc', prevLOC + fileLa - fileLd)
                setattr(prevFileChanged, 'authors', authors)
                setattr(prevFileChanged, 'lastchanged', unixTimeStamp)
                setattr(prevFileChanged, 'nuc', file_nuc)
            else:
                if(author not in authors):
                    authors.append(author)
                if(unixTimeStamp not in fileAges):
                    fileAges.append(unixTimeStamp)
                fileObject = CommitFile(fileName, fileLa - fileLd, authors, unixTimeStamp)
                commitFiles[fileName] = fileObject
            locModifiedPerFile.append(totalModified) 
            totalLOCModified += totalModified
            fileDirs = fileName.split("/")
            if( len(fileDirs) == 1 ):
                subsystem = "root"
                directory = "root"
            else:
                subsystem = fileDirs[0]
                directory = "/".join(fileDirs[0:-1])
            if( subsystem not in subsystemsSeen ):
                subsystemsSeen.append( subsystem )
            if( author in devExperience ):
                experiences = devExperience[author]
                exp += sum(experiences.values())
                if( subsystem in experiences ):
                    sexp = experiences[subsystem]
                    experiences[subsystem] += 1
                else:
                    experiences[subsystem] = 1
                try:
                    rexp += (1 / (age) + 1)
                except:
                    rexp += 0
            else:
                devExperience[author] = {subsystem: 1}
            if( directory not in directoriesSeen ):
                directoriesSeen.append( directory )
            la += fileLa
            ld += fileLd
            nf += 1
            filesSeen += fileName + ",CAS_DELIMITER,"
        if( nf < 1):
            return ""
        ns = len(subsystemsSeen)
        nd = len(directoriesSeen)
        ndev = len(authors)
        lt = lt / nf
        age = age / nf
        exp = exp / nf
        rexp = rexp / nf
        for fileLocMod in locModifiedPerFile:
            if (fileLocMod != 0 ):
                avg = fileLocMod/totalLOCModified
                entrophy -= ( avg * math.log( avg,2 ) )
        statProperties += ',"la":"' + str( la ) + '\"'
        statProperties += ',"ld":"' + str( ld ) + '\"'
        statProperties += ',"fileschanged":"' + filesSeen[0:-1] + '\"'
        statProperties += ',"nf":"' + str( nf ) + '\"'
        statProperties += ',"ns":"' + str( ns ) + '\"'
        statProperties += ',"nd":"' + str( nd ) + '\"'
        statProperties += ',"entrophy":"' + str(  entrophy ) + '\"'
        statProperties += ',"ndev":"' + str( ndev ) + '\"'
        statProperties += ',"lt":"' + str( lt ) + '\"'
        statProperties += ',"nuc":"' + str( nuc ) + '\"'
        statProperties += ',"age":"' + str( age ) + '\"'
        statProperties += ',"exp":"' + str( exp ) + '\"'
        statProperties += ',"rexp":"' + str( rexp ) + '\"'
        statProperties += ',"sexp":"' + str( sexp ) + '\"'
        return statProperties

    def clone(self, repo):
        repo_dir = os.chdir(os.path.dirname(__file__) + self.REPO_DIRECTORY)
        logging.info('Git cloning repo: '+ str(repo) )
        cloneResult = str(subprocess.check_output(
                  self.CLONE_CMD.format(repo.url, './' + repo.id),
                  shell= True,
                  cwd = repo_dir ) )
        logging.info('Done cloning.')
        return True
    def pull(self, repo):
        repo_dir = os.path.dirname(__file__) + self.REPO_DIRECTORY + repo.id
        subprocess.call(self.RESET_CMD, shell=True, cwd= repo_dir)
        subprocess.call(self.CLEAN_CMD, shell=True, cwd= repo_dir)
        logging.info('Pulling latest changes from repo: '+ str(repo) )
        fetchResult = str(subprocess.check_output(
                  self.RESET_CMD + "\n" + self.PULL_CMD ,
                  shell=True,
                  cwd=  repo_dir  ) )
        logging.info('Done fetching.')
        return True

class MedianModel:
  def __init__(self, metrics, repo_id):
    self.metrics = metrics
    self.repo_id = repo_id
    self.psig = 0.05
  def buildModel(self):
    self.calculateMedians()
  def getMedian(self,metric):
    median_props = ""
    try:
      metric_buggy = getattr(self.metrics, metric + "_buggy")
      metric_nonbuggy = getattr(self.metrics, metric + "_nonbuggy")
      pvalue = self.wilcoxFn(robjects.FloatVector(metric_buggy), robjects.FloatVector(metric_nonbuggy))[2][0]
      buggy_median = self.medianFn(robjects.FloatVector(metric_buggy))
      nonbuggy_median = self.medianFn(robjects.FloatVector(metric_nonbuggy))
      median_props += '"' + metric + 'buggy":"' + str(buggy_median[0]) + '", '
      median_props += '"' + metric + 'nonbuggy":"' + str(nonbuggy_median[0]) + '", '
      if pvalue <= self.psig:
        median_props += '"' + metric + '_sig":"1", '
      else:
        median_props += '"' + metric + '_sig":"0", '
    except:
      logging.info("Metric " + metric + " could not be used in the median model for repo " + self.repo_id)
    return median_props
  def calculateMedians(self):
    metricObject = '"repo":"' + self.repo_id + '", '
    metricObject += self.getMedian("ns")
    metricObject += self.getMedian("nd")
    metricObject += self.getMedian("nf")
    metricObject += self.getMedian("entrophy")
    metricObject += self.getMedian("la")
    metricObject += self.getMedian("ld")
    metricObject += self.getMedian("lt")
    metricObject += self.getMedian("ndev")
    metricObject += self.getMedian("age")
    metricObject += self.getMedian("nuc")
    metricObject += self.getMedian("exp")
    metricObject += self.getMedian("rexp")
    metricObject += self.getMedian("sexp")
    metricObject = metricObject[:-2]
    metricsSession = Session()
    metrics = Metrics(json.loads('{' + metricObject + '}'))
    metricsSession.merge(metrics)
    metricsSession.commit()

class GithubIssueTracker:
	owner = None											
	repo = None												
	request_repos = "https://api.github.com/repos"			
	request_auth = "https://api.github.com/authorizations" 	
	def __init__(self, owner, repo):
		self.owner = owner
		self.repo = repo
		self.auth_token = None
		self.authenticate() 
	def authenticate(self):
		s = requests.Session()
		username = config["github"]["user"]
		password = config["github"]["pass"]
		s.auth = (username, password)
		payload = {"scopes": ["repo"]}
		r = s.get(self.request_auth, params=payload)
		if r.headers.get('x-ratelimit-remaining') == '0':
			logging.info("Github quota limit hit -- waiting")
			while r.headers.get('x-ratelimit-remaining') == '0':
				time.sleep(600) 
				r = s.get(self.request_auth, params=payload)
				data = r.json()
		data = r.json()[0]
		if r.status_code >= 400:
			msg = data.get('message')
			logging.error("Failed to authenticate issue tracker: \n" +msg)
			return 
		else:
			self.auth_token = data.get("token")
			requests_left = r.headers.get('x-ratelimit-remaining')
			logging.info("Analyzer has " + requests_left + " issue tracker calls left this hour")
	def getDateOpened(self, issueNumber):
		logging.info("searching for issue: " + str(issueNumber))
		logging.info(self.request_repos + "/" + self.owner + "/" +
				self.repo + "/issues/" + issueNumber)
		header = {'Authorization': 'token ' + self.auth_token}
		r = requests.get(self.request_repos + "/" + self.owner + "/" +
				self.repo + "/issues/" + issueNumber, headers=header)
		data = r.json()
		if r.status_code == 403:
			if r.headers.get('x-ratelimit-remaining') == '0':
				logging.info("Github quota limit hit -- waiting")
				while r.headers.get('x-ratelimit-remaining') == '0':
					time.sleep(600) 
					r = requests.get(self.request_repos + "/" + self.owner + "/" +
						self.repo + "/issues/" + issueNumber, headers=header)
					data = r.json()
		elif r.status_code >= 400:
			msg = data.get('message')
			logging.info("issue not found")
			return None
		else:
			try:
				date = (dateutil.parser.parse(data.get('created_at'))).timestamp()
				return date
			except:
				logging.error("ISSUE TRACKER FAILURE: Could not get created_at from github issues API")
				return None

class GitCommitLinker:
  REPO_DIR = "ingester/CASRepos/git/" 
  def __init__(self, repoId):
    self.repo_path = os.path.join(os.path.dirname(__file__), '..', self.REPO_DIR + repoId)
    self.repo_id = repoId
  def linkCorrectiveCommits(self, corrective_commits, all_commits):
    linked_commits = {} 
    first_fixed_by = {} 
    for corrective_commit in corrective_commits:
      buggy_commits = self._linkCorrectiveCommit(corrective_commit)
      for buggy_commit in buggy_commits:
        if buggy_commit in linked_commits:
          linked_commits[buggy_commit].append(corrective_commit.commit_hash)
        else:
          linked_commits[buggy_commit] = [corrective_commit.commit_hash]
        if buggy_commit not in first_fixed_by or corrective_commit.author_date_unix_timestamp < first_fixed_by[buggy_commit].author_date_unix_timestamp:
          first_fixed_by[buggy_commit] = corrective_commit
      corrective_commit.linked = True 
    for commit in all_commits:
      if commit.commit_hash in linked_commits:
        commit.contains_bug = True
        commit.fixes = json.dumps(linked_commits[commit.commit_hash])
        commit.first_fixed_by = first_fixed_by[commit.commit_hash].commit_hash 
        commit.unix_timestamp_first_fix = first_fixed_by[commit.commit_hash].author_date_unix_timestamp 
  def _linkCorrectiveCommit(self, commit):
    region_chunks = self.getModifiedRegions(commit)
    logging.info("Linkage for commit " + commit.commit_hash)
    for k,v in region_chunks.items():
      logging.info("-- file: " + k)
      logging.info("---- loc modified: " + str(v))
    bug_introducing_changes = self.gitAnnotate(region_chunks, commit)
    return bug_introducing_changes
  def _getModifiedRegionsOnly(self, diff, files_modified):
    region_diff = {}
    list_ext_dir = os.path.join(os.path.dirname(__file__), "code_file_extentions.txt")
    file_exts_to_include = open(list_ext_dir).read().splitlines()
    for file in files_modified:
      if file != "'" and file != "":
        file_info = file.split(".")
        if len(file_info) > 1:
          file_ext = (file_info[1]).lower()
          if file_ext.upper() in file_exts_to_include:
            region_diff[file] = []
    regions = diff.split("diff --git")[1:] 
    for region in regions:
      chunks_initial = region.split(":CAS_DELIMITER_START:@@")
      if len(chunks_initial) == 1:
        continue
      file_info = chunks_initial[0] 
      file_info_split = file_info.split(" ")
      file_name = file_info_split[1][2:] 
      if file_name == None or file_name not in region_diff:
        continue
      for chunk in range(1, len(chunks_initial), 1):
        code_info_chunk = chunks_initial[chunk].split("@@",1) 
        line_info = code_info_chunk[0] 
        code_info = code_info_chunk[1] 
        mod_line_info = line_info.split(" ")[1] 
        mod_code_info = code_info.replace("\\n","").split(":CAS_DELIMITER:")[1:-1] 
        if mod_line_info[0] != '-':
          continue
        if mod_line_info.find(",") != -1:
          mod_line_info = mod_line_info[0:mod_line_info.find(",")]
        current_line = abs(int(mod_line_info)) 
        for section in mod_code_info:
          if section.startswith(":CAS_DELIMITER_START:-"):
            region_diff[file_name].append(str(current_line))
            current_line += 1 
    return region_diff
  def getModifiedRegions(self, commit):
    diff_cmd = "git diff " + commit.commit_hash + "^ "+ commit.commit_hash + " --unified=0 " \
      + ' | while read; do echo ":CAS_DELIMITER_START:$REPLY:CAS_DELIMITER:"; done'
    try:
      diff = str(subprocess.check_output(diff_cmd, shell=True, cwd= self.repo_path, executable="/bin/bash" ))
      diff_cmd_lines_changed = "git diff " + commit.commit_hash + "^ "+ commit.commit_hash + " --name-only"
      files_modified = str( subprocess.check_output( diff_cmd_lines_changed, shell=True, cwd= self.repo_path )).replace("b'", "").split("\\n")
      return self._getModifiedRegionsOnly(diff, files_modified)
    except:
      return {}
  def gitAnnotate(self, regions, commit):
    bug_introducing_changes = []
    for file, lines in regions.items():
      for line in lines:
        if line != 0 and line != "0" :
          buggy_change = str( subprocess.check_output( "git blame -L" + line + ",+1 " + commit.commit_hash + "^ -l -- '" \
                            + file + "'", shell=True, cwd= self.repo_path )).split(" ")[0][2:]
          if buggy_change not in bug_introducing_changes:
            bug_introducing_changes.append(buggy_change)
    return bug_introducing_changes

class BugFinder:
	def __init__(self, allCommits, correctiveCommits, issueTracker):
		self.allCommits = allCommits
		self.correctiveCommits = correctiveCommits
		self.issueTracker = issueTracker
	def findIssueOpened(self, correctiveCommit):
		issue_opened = None
		if(self.issueTracker is None or hasattr(self.issueTracker, "getDateOpened") == False):
			return None
		idMatch = re.compile('
		issue_ids = idMatch.findall(correctiveCommit.commit_message)
		issue_ids = [issue_id.strip('
		if len(issue_ids) > 0:
			issue_opened = self.issueTracker.getDateOpened(issue_ids[0])
			for issue_id in issue_ids:
				logging.info('Searching for issue id: ' + issue_id)
				curr_issue_opened = self.issueTracker.getDateOpened(issue_id)
				if curr_issue_opened is not None:
					if int(curr_issue_opened) < int(issue_opened):
						issue_opened = curr_issue_opened
		return issue_opened
	def searchForBuggyCommit(self, correctiveCommit):
		bug_introduced_prior = correctiveCommit.author_date_unix_timestamp
		issue_opened = self.findIssueOpened(correctiveCommit)
		if issue_opened is not None:
			bug_introduced_prior = issue_opened
		correctiveFiles = correctiveCommit.fileschanged.split(",CAS_DELIMITER,")
		for commit in self.allCommits:
			if int(commit.author_date_unix_timestamp) < int(bug_introduced_prior):
				commitFiles = commit.fileschanged.split(",CAS_DELIMITER,")
				for commitFile in commitFiles:
					if commitFile in correctiveFiles:
						return commit
		return -1 
	def markBuggyCommits(self):
		for correctiveCommit in self.correctiveCommits:
			buggyCommit = self.searchForBuggyCommit(correctiveCommit)
			if buggyCommit is not -1:
				buggyCommit.contains_bug = True

				.order_by( Commit.author_date_unix_timestamp.desc())

				)
	corrective_commits = (session.query(Commit)
				.filter( 
					( Commit.fix == "True" ) &

				.all()
				)

		git_commit_linker = GitCommitLinker(repo_id)

		repository_to_analyze.status = "Error"
		session.commit() 
		raise
	if repository_to_analyze.status != "Error":
		repository_to_analyze.status = "In Queue to Build Model"
		session.commit()

  def _isMetricSignificant(self, formula_metrics, metric):
    sig_column = 4
    if len(formula_metrics) == 0:
      formula = "is_buggy~" + metric
      fit = self.stats.glm(formula, data=self.data, family="binomial")
      summary = self.base.summary(fit)
      try:
        metric_sig = summary.rx2('coefficients').rx(2,sig_column)[0] 
        if metric_sig <= self.sig_threshold:
          return True
        else:
          return False
      except:
        return False
    else:
      num_metrics = len(formula_metrics)+2 
      formula = "is_buggy~" + "+".join(formula_metrics) + "+" + metric
      fit = self.stats.glm(formula, data=self.data, family="binomial")
      summary = self.base.summary(fit)
      try: 
        for row in range(2,num_metrics+1):
          metric_sig = summary.rx2('coefficients').rx(row,sig_column)[0]
          if metric_sig > self.sig_threshold:
            return False
        return True 
      except:
        return False
  def _buildModelIncrementally(self):
    metrics_list = ["la","ld","lt","ns","nd","nf","ndev","age","nuc","exp","rexp","sexp","entrophy"]
    formula_metrics = []
    current_dir = os.path.dirname(__file__)
    dir_of_datasets = current_dir + "/datasets/"
    self.data = self.readcsv(dir_of_datasets + self.repo_id + ".csv", header=True, sep = ",")
    for metric in metrics_list:
      if self._isMetricSignificant(formula_metrics, metric):
        formula_metrics.append(metric)
    self._storeCoefficients(formula_metrics)
    self.calculateCommitRiskyness(self.commits, formula_metrics)
  def _getCoefficients(self, formula_coefs):
    coef_dict = {} 
    formula = "is_buggy~" + "+".join(formula_coefs)
    fit = self.stats.glm(formula, data=self.data, family="binomial")
    for coef in formula_coefs:
      coef_dict[coef] = fit.rx2('coefficients').rx2(coef)[0]
    return coef_dict
  def _getInterceptValue(self, coefs):
    formula = "is_buggy~" + "+".join(coefs)
    fit = self.stats.glm(formula, data=self.data, family="binomial")
    summary = self.base.summary(fit)
    return summary.rx2('coefficients').rx(1)[0], summary.rx2('coefficients').rx(1,4)[0]
  def _getCoefficientObject(self, coef_name, coef_value):
    coef_object = ""
    coef_object += '"' + str(coef_name) + '":"' + str(coef_value)
    return coef_object + '",'
  def _storeCoefficients(self, coefficient_names):
    coefs = ""
    coefs += '"repo":"' + str(self.repo_id) + '",'
    if len(coefficient_names) == 0:
      coefficient_dict = {}
    else:
      coefficient_dict = self._getCoefficients(coefficient_names)
      intercept_value, intercept_pvalue = self._getInterceptValue(coefficient_names)
      if intercept_pvalue <= self.sig_threshold:
        intercept_sig = 1
      else:
        intercept_sig = 0
      coefs += self._getCoefficientObject("intercept", intercept_value)
      coefs += self._getCoefficientObject("intercept_sig", intercept_sig)
    sig_coefs = [] 
    all_coefs = ["ns", "nd", "nf", "entrophy", "la", "ld", "lt", "ndev", "age", "nuc", "exp", "rexp", "sexp"]
    for coef_name, coef_value in coefficient_dict.items():
      coefs += self._getCoefficientObject(coef_name, coef_value)
      coefs += self._getCoefficientObject(coef_name + "_sig", 1) 
      sig_coefs.append(coef_name)
    for c in all_coefs:
      if c not in sig_coefs:
        coefs += self._getCoefficientObject(c, -1) 
        coefs += self._getCoefficientObject(c + "_sig", 0)
    coefs = coefs[:-1]
    coefSession = Session()
    allCoef = GlmCoefficients(json.loads('{' + coefs + '}'))
    coefSession.merge(allCoef)
    coefSession.commit()
    coefSession.close()
  def calculateCommitRiskyness(self, commits, coefficient_names):
    if len(coefficient_names) == 0:
      coefficient_dict = {}
      model_available = False
    else:
      coefficient_dict = self._getCoefficients(coefficient_names)
      model_available = True
      intercept_value, intercept_pvalue = self._getInterceptValue(coefficient_names)
    for commit in commits:
      if model_available == False:
        commit.glm_probability = -1
      else:
        coefs_sum = 0
        for coef_name, coef_value in coefficient_dict.items():
          coefs_sum += (coef_value * getattr(commit, coef_name))
        try:
          riskyness = 1/(1+ math.exp(-intercept_value-coefs_sum))
        except OverflowError:
          logging.error("Overflow error for repo " + self.repo_id)
          logging.error("Calculating riskyness for " + commit.commit_hash)
          logging.error("Sum of coefficients: " + str(coefs_sum))
          logging.error("Coeffiecents: " + str(coefficient_dict))
          riskyness = 0.01
        commit.glm_probability = riskyness
